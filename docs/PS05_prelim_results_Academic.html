Data preparation

Libraries

library(tidyverse)
library(DT)

Annual electricty data

kwh_annual <- read.csv("./output/kwh_annual_2026-03-04.csv")
str(kwh_annual)

annual_academic <- kwh_annual %>%
  filter(type == "Academic") %>% 
  mutate(MTCO2e = round(kwh_corr*0.30082405, digits = 2), 
         Cost = round(kwh_corr*0.08138507,digits=2),
         kwh_sqft = round(kwh_corr/sqft, digits=2))%>%
  arrange(desc = TRUE) %>%
  select(meter, NAME, days_perc, kwh, kwh_corr, sqft, Cost, MTCO2e, kwh_sqft)
str(annual_academic)

Daily electricity data

kwh_daily <- read.csv("./output/kwh_daily_2026-03-04.csv")
str(kwh_daily)
academic_kwh_daily <- kwh_daily %>% 
  filter(type == "Academic") %>% 
  mutate(date = ymd(date), 
         month = month(date, label = TRUE), 
         day = wday(date, label = TRUE)) %>%
  mutate(kwh_sqft = kwh/sqft*365)

str(academic_kwh_daily)

Electricity intensity

academic_kwh_daily90 <- academic_kwh_daily %>%
  filter(days_perc >= 90)

ggplot(academic_kwh_daily90, aes(x = NAME, y = kwh_sqft, fill = NAME)) + 
         annotate("rect", xmin = -Inf, xmax = Inf, ymin = 7.4, ymax = 14.3, color = "lightgray", alpha = 0.3) + 
         geom_hline(yintercept = 10.3, linetype = "dashed", color = "white") + 
  geom_boxplot() +

  theme_bw() + 
  theme(legend.position = "none") +
