daily_date <- daily %>%
  mutate(date = mdy(date),
         month = month(date, label = TRUE),
         day = wday(date, label = TRUE)) %>%
  select(type, NAME, date, month, day, kwh, ave_temp)

daily_main <- filter(daily_date, NAME == "Main Meter", )
